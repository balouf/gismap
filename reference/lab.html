Examples

>>> from gismap.lab import ListLab
>>> lab = ListLab(author_list=['Tixeuil Sébastien', 'Mathieu Fabien'], name='mini')
>>> lab.update_authors()
>>> lab.update_publis()
>>> len(lab.authors)
2
>>> len(lab.publications)
435
>>> html = lab2graph(lab)
>>> html[:80]  
'\n<div id="mynetwork_..."></div>\n<div id="modal_..." class="modal">\n  <'
